import sys

sys.stdin = open('0_BOJ_재귀함수가뭔가요.txt', 'r')

#필요한 재귀함수들 정리
def repeat_front(num):
    for i in range(num):
        print('_'*i*4, end='')
        print('"재귀함수가 뭔가요?"')
        print('_'*i*4, end='')
        print('"잘 들어보게. 옛날옛날 한 산 꼭대기에 이세상 모든 지식을 통달한 선인이 있었어.')
        print('_'*i*4, end='')
        print('마을 사람들은 모두 그 선인에게 수많은 질문을 했고, 모두 지혜롭게 대답해 주었지.')
        print('_'*i*4, end='')
        print('그의 답은 대부분 옳았다고 하네. 그런데 어느 날, 그 선인에게 한 선비가 찾아와서 물었어."')

def repeat_rear(num):
    for i in range(num):
        print('_'*(num-i-1)*4, end='')
        print('라고 답변하였지.')

#출력하는 코드
num = int(input())

print('어느 한 컴퓨터공학과 학생이 유명한 교수님을 찾아가 물었다.')
repeat_front(num)

print('_'*4*num, end='')
print('"재귀함수는 자기 자신을 호출하는 함수라네"')

print('라고 답변하였지.')

repeat_rear(num)
